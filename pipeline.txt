pipeline {
    agent any

    stages {

        stage('PULL') {
            steps {
                git 'https://github.com/harshadaborse26/jenkins.git'
                echo 'Pull Success'
            }
        }

        stage('BUILD') {
            steps {
                dir('backend') {
                    sh 'mvn clean package -DskipTests'
                }
                echo 'BUILD SUCCESS'
            }
        }

        stage('TEST & SONAR') {
            steps {
                dir('backend') {
                    withSonarQubeEnv(credentialsId: 'SonarQube') {
                        sh '''
                        mvn sonar:sonar \
                          -Dsonar.projectKey=my-app \
                          -Dsonar.projectName="my app"
                        '''
                    }
                }
            }
        }
    }
}
